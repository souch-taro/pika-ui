# Taro-Router 路由库

# 注意

该项目已停止更新,请使用开源项目: https://github.com/Ke-Kou/mini-program-router-plus

# 说明

在原有的Taro路由功能上增加路由拦截, 参数非序列化传递, 页面传值功能

# 引入依赖

yarn add @souche-taro/router

# 使用

# config配置

import {initRouter} from "@souche-taro/router";

// 配置文件
interface Config {
    name: string, // 路由名称
    path: string, // 路由地址
    isTab: boolean, // 是否是tab页
    beforeEnter: (to, from, next) => void // 路由守卫
}

const config: Config = [
    {
        name: 'routerName',
        path: '/path/to/view',
        isTab: true,
        beforeEnter: (to, from, next) => {
        		next();
        } // 可以单独为某个路由配置前置守卫
    }
]

# 初始化

// 可以使用Taro.$$router
const router = initRouter({config})

# 全局路由守卫

// to将要进入的路由, from起始路由
// 需要调用next()才能跳转
router.beforeEach((to, from, next) => {
		next();
})


// 配置全局后置路由守卫
router.afterEach((to, from) => {})

# 跳转


// 跳转页面
// 可以使用的方法[push, replace, pop, switchTab, reLaunch, to]
Taro.$$router.push('routerName', {
    params: {}, // 路由参数
    // events: {}, // 事件监听(v2.7.3 https://developers.weixin.qq.com/miniprogram/dev/api/route/wx.navigateTo.html)
    success: () => {}, // 成功回调
    fail: () => {}, // 跳转失败回调
    complete: () => {}, // 成功/失败回调
})

// replace
Taro.$$router.replace('routerName', {
   ...same
})

// swicthTab
Taro.$$router.switchTab('routerName', {
   ...same
})

// to 用来替换replace和swicthTab
Taro.$$router.to('routerName', {
    ...same
})

// reLaunch
Taro.$$router.reLaunch('routerName', {
   ...same
})

// pop
Taro.$$router.pop('routerName', {
    ...same,
    delta: 1// 使用pop返回的页面数
})

# 获取路由参数

获取的参数是反序列化后的对象

import { getCurrentInstanceParams } from "@souche-taro/router";
const {shopCode, shareId} = getCurrentInstanceParams();

# a/b传值

emit参数适用于push, switchTab, to

// 添加监听事件
Taro.$$router.push('', {
    emit: {
        [eventName]: (data, next) => {
            console.log(data); // 111
            next(); // 执行promise.then中的事件
        }
    } 
})
// 执行监听事件
Taro.$$router.emit(eventName, 111).then(
    res => {
        // 执行监听后的事件
    }
)

// 由于无法监听小程序的后退按钮,因此需要在执行监听事件的类上增加unEmit注解
// unEmit的主要是监听componentWillUnmount事件,因此挂载在组件上会统一在当前页面销毁时执行
import {unEmit} from "@souche-taro/router";

@unEmit
export default class {
    method() {
        Taro.$$router.emit(name, 111)
    }
}
