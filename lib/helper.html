# Taro-helper 工具库

# 说明

一些常用的工具集合

# 快速上手

安装:

yarn add @souche-taro/helper

使用

import {utils, tools, decorators} from '@souche-taro/helper'

// 调用例子
tools.isFunction(fn)

utils.auth.hasAuth(utils.auth.AUTH_SCOPE_NAME.userInfo)

@decorators.bind

# auth模块

// 获取执行对应授权的返回值
export interface authInfo {
    success: boolean; // 是否获取成功
    data: any; // 获取的内容
    openAuthSetting: boolean; // 是否打开过设置面板
    error: null | Error; // 错误信息
}
// 权限获取附加参数
export interface authMeta {
    openSetting?: boolean; // 要求在二次请求的时候，当请求权限为denny的时候打开权限控制板
    openSettingForce?: boolean; // 请求权限为denny的时候强制要求打开权限控制板，忽略是否是第一次
    cache?: boolean; // 是否缓存当前权限数据
}
// 权限的枚举值
export declare const AUTH_SCOPE_NAME: {
    userInfo: string;
    userLocation: string;
    userLocationBackground: string;
    address: string;
    record: string;
    album: string;
    camera: string;
    bluetooth: string;
};
// 获取已经授权的内容
export declare function getAuthSetting(): Promise<Taro.getSetting.SuccessCallbackResult>;
// 打开授权面板
export declare function openAuthSetting(): Promise<Taro.openSetting.SuccessCallbackResult>;
// 判断是否拥有某个权限
export declare function hasAuth(authScope: string): Promise<boolean>;

// 下面是对应授权的调用方法
// https://developers.weixin.qq.com/miniprogram/dev/framework/open-ability/authorize.html#scope-%E5%88%97%E8%A1%A8
export declare const authUserInfo: (options: Taro.getUserInfo.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserGetLocation: (options: Taro.getLocation.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserOpenLocation: (options: Taro.openLocation.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserUpdateLocation: (options: Taro.startLocationUpdate.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserLocationBackground: (options: Taro.startLocationUpdate.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserChooseAddress: (options: Taro.chooseAddress.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserGetRecorderManager: (options: Taro.startRecord.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserSaveImageToPhotosAlbum: (options: Taro.saveImageToPhotosAlbum.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserSaveVideoToPhotosAlbum: (options: Taro.saveVideoToPhotosAlbum.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserScanCode: (options: Taro.scanCode.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserChooseImage: (options: Taro.chooseImage.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserChooseVideo: (options: Taro.chooseImage.Option, meta?: authMeta | undefined) => Promise<authInfo>;
export declare const authUserOpenBluetoothAdapter: (options: unknown, meta?: authMeta | undefined) => Promise<authInfo>;

# tools

用于一些轻工具,比如类型判断

// 变量是否未定义
export declare function isUndef(v: any): boolean;
// 变量是否定义
export declare function isDef(v: any): boolean;
// 变量的值是true
export declare function isOnlyTrue(v: any): boolean;
// 变量的值是false
export declare function isOnlyFalse(v: any): boolean;
// 是否是基本类型
export declare function isPrimitive(value: any): boolean;
// 是否是object对象类型
export declare function isObject(obj: any): boolean;
// 获取一个对象的类型的字符串表示形式
export declare function toRawType(value: any): any;
// 判断是否是纯object类型
export declare function isPlainObject(obj: any): boolean;
// 判断是否是正则
export declare function isRegExp(v: any): boolean;
// 判断是否是函数
export declare function isFunction(fn: any): boolean;
// 判断一个数是否是合法的数组索引
export declare function isValidArrayIndex(val: any): boolean;
// 判断对象属性值和给定的是否一致
export declare function isSafeEqual(target: any, property: any, value: any): boolean;
// 缓存函数
export declare function cached(fn: any): (str: any) => any;
// 一次执行函数
export declare function once(fn: any): () => any;
// 延迟执行
export declare function delay(delayTime?: number): Promise<null>;
