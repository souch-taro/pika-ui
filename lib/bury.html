# Taro-Bury 埋点库

# 说明

针对多平台的埋点库

# 使用方法

# 安装

// 老鲁班版本
yarn add @souche-taro/bury
// 新鲁班版本
yarn add @souche-taro/bury2

# 导入配置

# React版本

// 老鲁班版本
import Bury from '@souche-taro/bury'

Bury.load({
    platform: '',
    userTag?: '',
    autoUserTag: true,
    debug: false,
    env: 'production', //development, prepub, production
    beforeReport: (typeId, data, next) => {
        next();
    }
})

// 新鲁班版本
import Bury from '@souche-taro/bury2'

Bury.load({
    platform: 3 | 4,
    userTag?: '', // 就是uid
    umid?: '', // 友盟id
    autoUserTag: true,
    appName: '',
    projectId: '',
    debug: false,
    env: 'production', //development, prepub, production
    beforeReport: (typeId, data, next) => {
        next();
    }
})

Vue.use(vueBuryDirective, {

# 手动埋点

// 老鲁班
// 老鲁班会自动将bury实例挂载到Taro上
Taro.$bury.sendById(typeId, {
    // ...业务参数
})

// 新鲁班
// 新鲁班不会自动将bury实例挂载到Taro上,需要自己手动挂载
// 该方法用于兼容之前老版本,方便切换
bury.sendById(typeId, {
    // ...其他参数
    eventType: 'CLICK' | 'IMPRESSION' // 新鲁班适用
})

bury.customizeSend({
    buryId, 
    eventType, 
    attributes: {
        // ...业务参数
    }
})

# 指令埋点

// 有参数版
<text v-bury:bury_id.show.tap="buryParams">{{ msg }}</text>
// 无参数版
<text v-bury:bury_id.show.tap>{{ msg }}</text>

参数解释

  1. bury_id: 埋点id, 效果同buryParams中的buryId参数
  2. show: 元素显示时触发
  3. tap: 元素点击时触发
  4. buryParams: 携带业务参数

# userTag相关

// 设置userTag
Taro.$bury.setUserTag('新userTag');
// 获取userTag
Taro.$bury.getUserTag();

# 埋点开关

// 关闭
Taro.$bury.stopBury()
// 打开
Taro.$bury.beginBury()

# typeId和参数统一修改

通过beforeReport中传入next参数来修改typeId和传递参数

Bury.load({
    beforeReport: (typeId, data, next) => {
        next({
            typeId: 'self-id', // 可以修改原typeId
            shopCode: '000123', // 如果data中有shopCode会替换原来的shopCode, 没有会增加
        });
    }
})

# 修改config

bury.setConfig({
    appName: '',
})
