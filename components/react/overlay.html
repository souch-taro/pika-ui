# 引入

如果需要按需引入, 则:

import { PkOverlay } from '@souche-taro/pika-react/lib';

组件依赖的样式文件（仅按需引用时需要）

@import '~@souche-taro/pika-react/dist/style/components/overlay.scss';

# 代码演示

# 遮罩层

<ShowCom title='遮罩层'>
    <PkButton
        onClick={() => {
            this.setState({ visible1: true });
        }}
    >
        点击显示遮罩层
    </PkButton>
    <PkOverlay
        visible={this.state.visible1}
        onClick={() => {
            this.setState({
                visible1: false,
            });
        }}
    />
</ShowCom>

# 自定义 children 的遮罩层

<ShowCom title='自定义 children 的遮罩层'>
    <PkButton
        onClick={() => {
            this.setState({ visible2: true });
        }}
    >
        点击显示带自定义 children 的遮罩层
    </PkButton>
    <PkOverlay
        visible={this.state.visible2}
        onClick={() => {
            this.setState({
                visible2: false,
            });
        }}
    >
        <View
            style={{
                width: '200px',
                height: '200px',
                margin: '100px auto 0 auto',
                backgroundColor: '#fff',
            }}
        />
    </PkOverlay>
</ShowCom>

# API

# PkOverlay Props

参数 类型 默认值 必填 说明
children React.ReactNode false 插槽
maskClosable boolean true false 点击蒙层是否可关闭蒙层
visible boolean false false 蒙层是否显示
zIndex number 8000 false 蒙层层级
onClick (e: CommonEvent) => void false 点击蒙层时触发，返回事件对象 Event
onClose ()=>void false 关闭蒙层时触发
