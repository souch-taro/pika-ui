        onClick={(): void => {
            this.setState({ visible1: true });
        }}
    >
        点击显示单行文本提示
    </PkButton>
    <PkToast
        text='这里是提示'
        visible={this.state.visible1}
        duration={2000}
        onClose={this.handleHideToast1}
    />

# API

# PkDialog Props

参数 类型 默认值 必填 说明
children React.ReactNode false 插槽
duration number false 提示持续时间，不传则不会自动关闭
icon React.ReactNode false 提示图标
text string false 提示文本
type 'succeed'|'failed'|'loading' false 提示类型
visible boolean false false 是否显示提示
showMask boolean true false 是否显示遮罩层
zIndex number 9000 false 提示层级
onClose () => void false 关闭提示时触发
